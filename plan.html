🔹 3.1 ServiceNow Script Include (Source)

javascript
var CaseUtils = Class.create();
CaseUtils.prototype = {
  createCase: function(data) {
    if (!data.short_description) {
      throw "Short description required";
    }
    var gr = new GlideRecord('x_case');
    gr.initialize();
    gr.short_description = data.short_description;
    gr.state = 'NEW';
    gr.insert();
    return gr.sys_id;
  }
};

🔹 3.2 Generated Java Service

java
@Service
@Slf4j
public class CaseService {

    private final CaseRepository repository;

    public CaseService(CaseRepository repository) {
        this.repository = repository;
    }

    @Transactional
    public UUID createCase(CreateCaseRequest request) {
        if (request.getShortDescription() == null) {
            throw new IllegalArgumentException("Short description required");
        }

        CaseEntity entity = new CaseEntity();
        entity.setShortDescription(request.getShortDescription());
        entity.setState(CaseState.NEW);

        CaseEntity saved = repository.save(entity);
        log.info("Case created with id {}", saved.getId());

        return saved.getId();
    }
}
